有限状态机 Finite System Manager

代码：

//状态机：
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// 
/// 人物行为
/// 
public enum Transition
{
    NullTransition = 0,
    SawPlayer,
    LostPlayer,
}

/// 
/// 人物状态
/// 
public enum StateID
{
    NullStateID = 0,
    ChasingPlayer,
    FollingPath,
}


public abstract class FSMState
{
    protected Dictionary map = new Dictionary();
    protected StateID stateID;
    public StateID ID { get { return stateID; } }

    /// 
    ///添加行为到map中
    /// 
    /// 
    /// 
    public void AddTransition(Transition trans, StateID id)
    {
        if (map.ContainsKey(trans))
        {
            Debug.LogError("此状态已存在");
            return;
        }
        map.Add(trans, id);
    }

    /// 
    /// 从map里删除行为
    /// 
    /// 
    /// 
    public void DeleteTransition(Transition trans, StateID id)
    {
        if (map.ContainsKey(trans))
        {
            map.Remove(trans);
            return;
        }
        Debug.LogError("此状态不存在");

    }


    /// 
    /// 获取当前状态
    /// 
    /// 
    /// 
    public StateID GetOutputState(Transition trans)
    {
        if (map.ContainsKey(trans))
        {
            return map[trans];
        }
        return StateID.NullStateID;
    }


    public virtual void DoBeforeEntering() { }
    public virtual void DobeforeLeaving() { }
    public abstract void Reason(GameObject anim);
    public abstract void Act(GameObject anim);

}


public class FSMSystem
{
    private List states=new List();
    public List States { get { return states; } }

    private StateID currentStateID;
    public StateID CurrentStateID { get { return currentStateID; } }

    private FSMState currentState;
    public FSMState CurrentState { get { return currentState; } }


    /// 
    /// 添加状态机系统里的状态
    /// 
    /// 
    public void AddState(FSMState s)
    {
        if (s == null)
        {
            Debug.LogError("FSMState为空引用");
            return;
        }


        if (states.Count == 0)
        {
            states.Add(s);
            currentState = s;
            currentStateID = s.ID;
            return;
        }


        foreach (FSMState state in states)
        {
            if (state.ID == s.ID)
            {
                Debug.LogError("此FSMState已存在");
                return;
            }
        }

        states.Add(s);
    }

    /// 
    /// 删除状态机系统利地状态
    /// 
    /// 
    public void DeleteState(StateID id)
    {
        if (id == StateID.NullStateID)
        {
            Debug.LogError("StateID为空");
            return;
        }

        foreach (FSMState state in states)
        {
            if (state.ID == id)
            {
                states.Remove(state);
                return;
            }
        }
        Debug.LogError("没有所查找的FSMState，无法删除");
    }



    /// 
    /// 执行行为
    /// 
    /// 
    public void PerformTransition(Transition trans)
    {
        if (trans == Transition.NullTransition)
        {
            Debug.LogError("行为为空，无法执行");
            return;
        }

        StateID id = currentState.GetOutputState(trans);
        if (id == StateID.NullStateID)
        {
            Debug.LogError("状态为空，无法执行");
            return;
        }

        currentStateID = id;

        foreach (FSMState state in states)
        {
            if (state.ID == currentStateID)
            {
                currentState.DobeforeLeaving();
                currentState = state;
                currentState.DoBeforeEntering();
                break;
            }

        }
    }

}



状态机的使用：
public class NPCController : MonoBehaviour
{

    public void SetTransition(Transition t) { fsm.PerformTransition(t); }

    private void Awake()
    }

    // Start is called before the first frame update
    void Start()
    {
        MakeFSM();

    }

    // Update is called once per frame
    void Update()
    {
        fsm.CurrentState.Reason(aim);
        fsm.CurrentState.Act(aim);        
    }



    private void MakeFSM()
    {        
        FollowPathState follow = new FollowPathState();
        follow.AddTransition(Transition.SawPlayer, StateID.ChasingPlayer);

        ChasePlayer chase = new ChasePlayer();
        chase.AddTransition(Transition.LostPlayer, StateID.FollingPath);

        fsm = new FSMSystem();
        fsm.AddState(follow);
        fsm.AddState(chase);        
    }

}


public class FollowPathState :FSMState
{
    public FollowPathState()
    {
        stateID = StateID.FollingPath;
    }

    public override void Reason(GameObject aim)
    {
        if (aim.GetComponent().RNum <= 5)
        {
            aim.GetComponent().SetTransition(Transition.SawPlayer);
        }
    }

    public override void Act(GameObject aim)
    {

        Debug.Log(aim.GetComponent().velocity);
        aim.GetComponent().velocity = new Vector2(-1, 0);
    } 
}


public class ChasePlayer : FSMState
{
    public ChasePlayer()
    {
        stateID = StateID.ChasingPlayer;
    }

    public override void Act(GameObject aim)
    {
        aim.GetComponent().velocity = new Vector2(1, 0);
    }

    public override void Reason(GameObject aim)
    {
        if (aim.GetComponent().RNum>5)
        {
            aim.GetComponent().SetTransition(Transition.LostPlayer);
        }
    }
}
